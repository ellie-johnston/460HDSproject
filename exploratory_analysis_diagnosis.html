from tabulate import tabulate
#plots 3 data sets together and gives table of their mean and max los in hrs
def plot_3(data1,data2,data3):

    fig = plt.figure()
    ax = fig.add_subplot(111)
    ax.plot(data1.spell_los_hrs, color ='orange', )
    ax.plot(data2.spell_los_hrs, color = 'red')
    ax.plot(data3.spell_los_hrs, 'green')
    plt.ylabel('length of stay in hours')
    plt.xlabel('Index')

    means = [[data1.spell_los_hrs.mean(), data1.spell_los_hrs.max()] , [data2.spell_los_hrs.mean(), data2.spell_los_hrs.max() ], 
    [data3.spell_los_hrs.mean(),data3.spell_los_hrs.max() ] ]
    print(tabulate(means, headers= ['Mean los hrs','Max los hours']))

print(plot_3(k_digestive,i_circulatory,j_respiratory))
